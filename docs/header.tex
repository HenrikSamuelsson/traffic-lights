
% --- List-like rendering for Doxygen "Parameters" (no borders, no blank column) ---
\usepackage{array}
\makeatletter
\newlength{\ParamNameCol}
\setlength{\ParamNameCol}{0.28\linewidth}
\renewenvironment{DoxyParams}[1]{%
  \par\vspace{0.5\baselineskip}%
  \noindent\textbf{#1}\par
  % Suppress \hline tokens Doxygen emits after each row
  \begingroup\let\hline\relax
  \noindent\begin{tabular}{@{}>{\itshape}p{\ParamNameCol} @{\quad}
                        >{\raggedright\arraybackslash}p{\dimexpr\linewidth-\ParamNameCol-2\tabcolsep\relax}@{}}
}{%
  \end{tabular}\endgroup\par\vspace{0.5\baselineskip}%
}
\makeatother
